\innerblock{Vector Space}{
A set $V$ over a field $F$ with a binary operation $+$ is a vector space if:
\begin{enumerate}
  \item $+$ is commutative
  \item $+$ is associative
  \item $+$ has identity $\vec{0}$
  \item every $\vec{x} \in V$ has inverse for $+$, written $-\vec{x}$
  \item $\alpha (\vec{x} + \vec{y}) = \alpha \vec{x} + \alpha \vec{y}$
  \item $(\alpha + \beta)\vec{x} = \alpha \vec{x} + \beta \vec{x}$
  \item $(\alpha \beta)\vec{x} = (\alpha) (\beta\vec{x})$
  \item $1 \vec{x} = \vec{x}$
\end{enumerate}
}
