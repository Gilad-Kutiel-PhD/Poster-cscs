\innerblock{Linear Map}{
Let $\mathbf{V}$ and $\mathbf{W}$ be vector spaces over the same field $\mathbf{K}$.
A function $f:\mathbf{V} \to \mathbf{W}$ is a linear map if 
for any two vectors $\mathbf{u}, \mathbf{v} \in \mathbf{V}$ and any scalar $c \in \mathbf{K}$:
\begin{enumerate}
  \item $f(\mathbf{u} + \mathbf{v}) = f(\mathbf{u}) + f(\mathbf{v})$
  \item $f(c \mathbf{u}) = c f(\mathbf{u})$
\end{enumerate}
}
